1 var
  2   WebElement;
  3 
  4 /**
  5  * @constructor
  6  */
  7 WebElement = function (elementId, wd) {
  8   /**
  9    * Get element ID for current session.
 10    *
 11    * @returns {String} Element ID.
 12    */
 13   this.getElementId = function () {
 14     return elementId;
 15   };
 16 
 17   /**
 18    * Click on an element.
 19    *
 20    * @returns {*}
 21    */
 22   this.click = function () {
 23     wd.clickElement(this);
 24 
 25     return this;
 26   };
 27 
 28   /**
 29    * Submit a FORM element.
 30    *
 31    * The submit command may also be applied to any element that is a descendant of a FORM element.
 32    *
 33    * @returns {*}
 34    */
 35   this.submit = function () {
 36     wd.submitElement(this);
 37 
 38     return this;
 39   };
 40 
 41   /**
 42    * Returns the visible text for the element.
 43    *
 44    * @returns {String}
 45    */
 46   this.getText = function () {
 47     return wd.getElementText(this);
 48   };
 49 
 50   /**
 51    * Send a sequence of key strokes to the active element. This command is similar to the WebDriver.type command in
 52    * every aspect except the implicit termination: The modifiers are not released at the end of the call. Rather, the
 53    * state of the modifier keys is kept between calls, so mouse interactions can be performed while modifier keys are
 54    * depressed.
 55    *
 56    * @param {String} value
 57    *
 58    * @returns {*}
 59    */
 60   this.type = function (value) {
 61     wd.type(this, value);
 62 
 63     return this;
 64   };
 65 
 66   /**
 67    * Query for an element's tag name.
 68    *
 69    * @returns {String}
 70    */
 71   this.getTagName = function () {
 72     return wd.getElementTagName(this);
 73   };
 74 
 75   /**
 76    * Clear a TEXTAREA or text INPUT element's value.
 77    *
 78    * @returns {*}
 79    */
 80   this.clear = function () {
 81     wd.clear(this);
 82 
 83     return this;
 84   };
 85 
 86   /**
 87    * Determine if an OPTION element, or an INPUT element of type checkbox or radiobutton is currently selected.
 88    *
 89    * @returns {Boolean}
 90    */
 91   this.isSelected = function () {
 92     return wd.isSelected(this);
 93   };
 94 
 95   /**
 96    * Determine if an element is currently enabled.
 97    *
 98    * @returns {Boolean}
 99    */
100   this.isEnabled = function () {
101     return wd.isEnabled(this);
102   };
103 
104   /**
105    * Determine if an element is currently displayed.
106    *
107    * @returns {Boolean}
108    */
109   this.isDisplayed = function () {
110     return wd.isDisplayed(this);
111   };
112 
113   /**
114    * Get the value of an element's attribute.
115    *
116    * @param {String} name
117    *
118    * @returns {String}
119    */
120   this.getAttribute = function (name) {
121     return wd.getAttribute(this, name);
122   };
123 
124   /**
125    * Determine an element's location on the page. The point (0, 0) refers to the upper-left corner of the page.
126    *
127    * @returns {x, y}
128    */
129   this.getLocation = function () {
130     return wd.getLocation(this);
131   };
132 
133   /**
134    * Determine an element's location on the screen once it has been scrolled into view.
135    *
136    * Note: This is considered an internal command and should only be used to determine an element's location for
137    *       correctly generating native events.
138    *
139    * @returns {x, y} The X and Y coordinates for the element on the page.
140    */
141   this.getLocationInView = function () {
142     return wd.getLocationInView(this);
143   };
144 
145   /**
146    * Determine an element's size in pixels.
147    *
148    * @returns {width, height}
149    */
150   this.getSize = function () {
151     return wd.getElementSize(this);
152   };
153 
154   /**
155    * Query the value of an element's computed CSS property. The CSS property to query should be specified using the CSS
156    * property name, not the JavaScript property name (e.g. background-color instead of backgroundColor).
157    *
158    * @param {String} propertyName
159    *
160    * @returns {String}
161    */
162   this.getCssProperty = function (propertyName) {
163     return wd.getCssProperty(this, propertyName);
164   };
165 
166   /**
167    * Start dragging current element and drop it over the second element.
168    *
169    * @param {WebElement} element Second element.
170    *
171    * @returns {*}
172    */
173   this.dragAndDropTo = function (element) {
174     wd.moveTo(this);
175     wd.buttonDown();
176     wd.moveTo(element);
177     wd.buttonUp();
178 
179     return this;
180   };
181 };
182 
183 module.exports = WebElement;
