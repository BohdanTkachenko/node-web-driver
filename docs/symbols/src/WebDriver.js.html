 67   /**
 68    * Throw error by error code.
 69    *
 70    * @private
 71    *
 72    * @param {Number} code Error code.
 73    * @param {String} [msg] Text to omit default error message.
 74    *
 75    * @throws {AbstractError}
 76    */
 77   function _errorByCode (code, msg) {
 78     var
 79       dir = __dirname + '/errors/',
 80       errors = fs.readdirSync(dir),
 81       i, fn,
 82       error, obj;
 83 
 84     for (i = 0; i < errors.length; i++) {
 85       fn = errors[i];
 86 
 87       if (!/\.js$/.test(fn)) {
 88         continue;
 89       }
 90 
 91       error = require(dir + fn);
 92       obj = new error(msg);
 93 
 94       if (obj.code !== code) {
 95         continue;
 96       }
 97 
 98       throw obj;
 99     }
100 
101     throw new UnknownError(msg);
102   }

104   /**
105    * Make a request to webdriver with given params
106    *
107    * @private
108    *
109    * @param {String} [path] webdriver path (method) to call.
110    * @param {String} [method] HTTP method for request (e.g. GET, POST, PUT, ...).
111    * @param {String} [data] Data for POST payload
112    *
113    * @return {Object} Request result.
114    */
115   function _request (path, method, data) {
116     method = method || 'GET';
117     data = data || null;
118 
119     if (/:sessionId/.test(path)) {
120       if (!session.sessionId) {
121         throw new NoSuchDriver();
122       }
123 
124       path = path.replace(':sessionId', session.sessionId);
125     }
126 
127     var req, res, resObj;
128 
129     req = http.request({
130       protocol: remoteDriverHttpParams.protocol,
131       port: remoteDriverHttpParams.port,
132       host: remoteDriverHttpParams.host,
133       method: method,
134       path: path,
135       headers: {},
136       body: data
137     });
138 
139     res = req.end();
140 
141     res.statusCode = parseInt(res.statusCode, 10);
142 
143     if (res.statusCode === 303) {
144       if (!res.headers.location) {
145         throw UnknownError;
146       }
147 
148       return res.headers.location;
149     } else if (res.statusCode === 404) {
150       throw new UnknownCommand();
151     } else if (res.statusCode !== 200) {
152       throw new UnknownError('Unknown status code: ' + req.statusCode);
153     }
154 
155     if (res.headers['content-type'] !== 'application/json; charset=utf-8') {
156       throw method + ' ' + path + ': got content-type ' + res.headers['content-type'] +
157         '. should be "application/json; charset=utf-8"';
158     }
159 
160     try {
161       resObj = JSON.parse(res.body.toString());
162     } catch (e) {
163       throw method + ' ' + path + ': cannot parse JSON';
164     }
165 
166     if (resObj.status > 0) {
167       _errorByCode(resObj.status, (resObj.value && resObj.value.message) ? resObj.value.message : null);
168     }
169 
170     return resObj;
171   }

495   /**
496    * Search for an element on the page, starting from the identified element.
497    *
498    * <p>The located element will be returned as a WebElement object.<p>
499    *
500    * <p>The table below lists the locator strategies that each server should support.
501    * Each locator must return the first matching element located in the DOM.</p>
502    *
503    * <ul>
504    * <li><strong>class name</strong> — Returns an element whose class name contains the search value;
505    *                                         compound class names are not permitted</li>
506    * <li><strong>css selector</strong> — Returns an element matching a CSS selector</li>
507    * <li><strong>id</strong> — Returns an element whose ID attribute matches the search value</li>
508    * <li><strong>name</strong> — Returns an element whose NAME attribute matches the search value</li>
509    * <li><strong>link text</strong> — Returns an anchor element whose visible text matches the search value</li>
510    * <li><strong>partial link text</strong> — Returns an anchor element whose visible text partially matches
511    *                                                the search value</li>
512    * <li><strong>tag name</strong> — Returns an element whose tag name matches the search value</li>
513    * <li><strong>xpath</strong> — Returns an element matching an XPath expression.
514    *                                    The provided XPath expression must be applied to the
515    *                                    server "as is"; if the expression is not relative to the element root,
516    *                                    the server should not modify it.
517    *                                    Consequently, an XPath query may return elements not contained in the root
518    *                                    element's subtree</li>
519    *
520    * @param {String} using The locator strategy to use.
521    * @param {String} value The search target.
522    *
523    * @returns {WebElement} Located element.
524    */
525   this.getElement = function (using, value) {
526     var res = _post('/session/:sessionId/element', {
527       using: using,
528       value: value
529     });
530

532       throw new UnknownError();
533     }

643    * <p>The server must process the key sequence as follows:</p>

645    * <li>- Each key that appears on the keyboard without requiring modifiers are sent as a keydown followed by a key up.</li>

990   /**
991    * Polls specified function.
992    *
993    * @param conditionFn Function to poll. It should return true or false.
994    * @param thisArg
995    * @param {Number} [timeout] Timeout in seconds. Defaults to 15 s.
996    * @param {Number} [interval] Interval in seconds. Defaults to 0.5 s.
997    * @returns {Boolean}
998    */
999   this.waitForCondition = function (conditionFn, thisArg, timeout, interval) {
1000     if (typeof conditionFn !== 'function') {
1001       throw new TypeError('condition must be a function that returns a Boolean');
1002     }
1003 
1004     timeout = timeout || 15;
1005     interval = interval || 0.5;
1006 
1007     var timeLeft = timeout;
1008     while (timeLeft > 0) {
1009       if (conditionFn.call(thisArg || this)) {
1010         return true;
1011       }
1012 
1013       timeLeft -= interval;
1014       _sleep(interval);
1015     }
1016 
1017     return false;
1018   };

1020   /**
1021    * Wait for element and return it when it appears.
1022    *
1023    * @param {String} using See WebDriver.getElement() for details.
1024    * @param {String} value Value for query.
1025    * @param {Number} [timeout] Timeout in seconds.
1026    * @param {Number} [interval] Interval in seconds.
1027    *
1028    * @returns {WebElement}
1029    */
1030   this.waitForElement = function (using, value, timeout, interval) {
1031     timeout = timeout || 15;
1032     interval = interval || 0.5;
1033 
1034     var conditionFn = function () {
1035       if (!this.hasElement(using, value)) {
1036         return false;
1037       }
1038 
1039       if (!this.getElement(using, value).isDisplayed()) {
1040         return false;
1041       }
1042 
1043       return true;
1044     };
1045 
1046     if (!this.waitForCondition(conditionFn, this, timeout, interval)) {
1047       throw new Timeout('Element "' + value + '" was not found using "' + using + '" after ' + timeout +
1048         ' seconds with polling interval of ' + interval + ' seconds.');
1049     }
1050 
1051     return this.getElement(using, value);
1052   };
